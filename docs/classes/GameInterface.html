    Class GameInterface

    GameInterface extends GameCanvas functionality with UI components. All UI components (Button, NumberInput, Slider, TopBar, BottomBar, UIBar) are available as static properties.

        A function to handle click events

      Returns number

      ID that can be used to remove handler with removeClickHandler

      Example: <caption>Make a drawing move whenever there is a click</caption>

      let xpos = 100;
      let ypos = 100;
      // Register a handler to update our variable each time
      // there is a click.
      game.addClickHandler(
          function ({x,y}) {
            // set variables...
            xpos = x;
            ypos = y;
          }
      )
      // Now create a drawing that uses the variable we set.
      game.addDrawing(
          function ({ctx}) {ctx.fillRect(xpos,ypos,30,30)}
      )

        draw function OR an object with a draw callback method

      Returns number

      ID that can be used in removeDrawing callback to remove drawing.

      Example: <caption>Passing a draw function</caption>

      game.addDrawing(
          function ({ctx,elapsed}) {
              ctx.beginPath();
              ctx.moveTo(200,200);
              ctx.lineTo(100,200+Math.sin(elapsed/10)*200);
              ctx.stroke();
          }
      );

      Example: <caption>Passing an object with a draw method</caption>

      game.addDrawing(
           { x : 0,
             y : 0,
             w : 100,
             h : 100,
             draw ({ctx,stepTime,width,height}) {
                this.x += stepTime/20;
                this.y += stepTime/20;
                if (this.x > width) { this.x = 0 }
                if (this.y > height) { this.y = 0 }
                ctx.fillRect(this.x,this.y,this.w,this.h)
             },
           }
      );

      Example: <caption>A drawing that will remove itself when it leaves the screen</caption>

      game.addDrawing(
          function ({ctx,elapsed,width,remove}) {
              const x = elapsed / 20
              ctx.fillRect(x,20,20,20);
              if (x > width) { remove() }
          }
      );
